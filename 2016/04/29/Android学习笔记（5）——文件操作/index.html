读写文件

在Android中文件的读写用的是java中的file类，所以读写操作和java中是一样的。

  • 写入操作

    String data = editText.getText().toString();
FileOutputStream fileOutputStream = null;
BufferedWriter bufferedWriter = null;
try {
    fileOutputStream = openFileOutput("date", Context.MODE_PRIVATE);
    bufferedWriter = new BufferedWriter(new OutputStreamWriter(fileOutputStream));
    bufferedWriter.write(data);
    //Toast.makeText(this, "shuru "+data, Toast.LENGTH_SHORT).show();
} catch (IOException e) {
    e.printStackTrace();
} finally {
    try {
        if (bufferedWriter != null) {
            bufferedWriter.close();
        }
    } catch (IOException e) {
        e.printStackTrace();
    }
}
  • 读出操作
    FileInputStream fileInputStream = null;
    BufferedReader bufferedReader = null;
    StringBuilder content = new StringBuilder();
    try {
        fileInputStream = openFileInput("date");
        bufferedReader = new BufferedReader(new InputStreamReader(fileInputStream));
        String line = "";
        while ((line = bufferedReader.readLine()) != null) {
            content.append(line);
        }
        textView.setText(content.toString());
        //Toast.makeText(this, "输出 "+content.toString(), Toast.LENGTH_SHORT).show();
    } catch (IOException e) {
        e.printStackTrace();
    } finally {
        if (bufferedReader != null) {
            try {
                bufferedReader.close();
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
    }

SQLite

在Android中使用数据库通过继承SQLiteOpenHelper类进行数据库的操作
public class SQLiteHelper extends SQLiteOpenHelper{

    public static final String CREATE_ADDRESS_BOOK="create table address_book ("
            + "id integer primary key autoincrement,"
            + "name text not null,"
            + "phone_number text not null)";

    private Context context;

    public SQLiteHelper(Context context, String name, CursorFactory factory, int version) {
        super(context, name, factory, version);
        this.context=context;
    }

    @Override
    public void onCreate(SQLiteDatabase db) {//数据库创建，只会在软件安装第一次启动时执行，之后更新软件不执行
        db.execSQL(CREATE_ADDRESS_BOOK);
        Toast.makeText(context, "创建数据库文件完成！", Toast.LENGTH_SHORT).show();
    }

    @Override
    public void onUpgrade(SQLiteDatabase db, int oldVersion, int newVersion) {//更新软件后第一次启动执行
        db.execSQL("dorp table if exists address_book");
        onCreate(db);
    }
}

具体操作数据库可以单独写一个dao类（我开发j2ee的习惯）不知道在安卓中这样规范不？ 

  public class AddressBookDao {
    private SQLiteHelper helper;
    private SQLiteDatabase db;

    public AddressBookDao(Context context) {
        helper = new SQLiteHelper(context, "AddressBook", null, 1);//实例化helper
        db = helper.getWritableDatabase();//获得db
    }

    public List<AddressBook> getAll() {
        List<AddressBook> addressBooks = new ArrayList<AddressBook>();
        AddressBook addressBook = null;
        Cursor cursor = db.query("address_book", null, null, null, null, null, null);
        if (cursor.moveToFirst()) {
            do {
                // 遍历Cursor对象，取出数据并打印
                addressBook=new AddressBook();
                int id = cursor.getInt(cursor.getColumnIndex("id"));
                addressBook.setId(Integer.valueOf(id));
                String name = cursor.getString(cursor.getColumnIndex("name"));
                addressBook.setName(name);
                String phone_number = cursor.getString(cursor.getColumnIndex("phone_number"));
                addressBook.setPhone_number(phone_number);
                addressBooks.add(addressBook);
            } while (cursor.moveToNext());
        }
        cursor.close();
        return addressBooks;
    }

    public List<AddressBook> findByName(String filter_name) {
        Log.d("search", filter_name);
        List<AddressBook> addressBooks = new ArrayList<AddressBook>();
        AddressBook addressBook = null;
        Cursor cursor = db.query("address_book", null, "name like ? or phone_number like ?", new String[]{"%"+filter_name+"%","%"+filter_name+"%"}, null, null, null);
        if (cursor.moveToFirst()) {
            do {
                // 遍历Cursor对象，取出数据并打印
                addressBook=new AddressBook();
                int id = cursor.getInt(cursor.getColumnIndex("id"));
                addressBook.setId(Integer.valueOf(id));
                String name = cursor.getString(cursor.getColumnIndex("name"));
                addressBook.setName(name);
                String phone_number = cursor.getString(cursor.getColumnIndex("phone_number"));
                addressBook.setPhone_number(phone_number);
                addressBooks.add(addressBook);
            } while (cursor.moveToNext());
        }
        cursor.close();
        return addressBooks;
    }

    public void addOrUpdate(Integer id,String name, String phone_number) {
        ContentValues values = new ContentValues();
        values.put("name", name);
        values.put("phone_number",phone_number);
        if(null==id){
            db.insert("address_book", null, values);
        }else{
            values.put("id", id.toString());
            db.update("address_book", values, "id=?", new String[]{id.toString()});
        }
    }


    public void deleteById(String id) {
        db.execSQL("delete from address_book where id=?",new String[]{id});
    }
}
