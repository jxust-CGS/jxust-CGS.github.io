操作Bitmap

/**  
 * 缩放图片
 * @param bitmap
 * @param width
 * @param height
 * @return
 */
  public Bitmap compression(Bitmap bitmap ,int width,int height){
      // 得到图片原始的高宽
      int rawHeight = bitmap.getHeight();
      int rawWidth = bitmap.getWidth();
      // 设定图片新的高宽
      int newWidth = width;
      int newHeight = height;
      // 计算缩放因子
      float heightScale = ((float) newHeight) / rawHeight;
      float widthScale = ((float) newWidth) / rawWidth;
      // 新建立矩阵
      Matrix matrix = new Matrix();
      matrix.postScale(heightScale, widthScale);
      // 设置图片的旋转角度
      //matrix.postRotate(-30);
      // 设置图片的倾斜
      //matrix.postSkew(0.1f, 0.1f);
      //将图片大小压缩
      //压缩后图片的宽和高以及kB大小均会变化
      Bitmap newBitmap = Bitmap.createBitmap(bitmap, 0, 0, rawWidth,rawWidth, matrix, true);
      // 将Bitmap转换为Drawable
      //Drawable newBitmapDrawable = new BitmapDrawable(newBitmap);
      //imageView.setImageDrawable(newBitmapDrawable);
      //然后将Bitmap保存到SDCard中,方便于原图片的比较
      //this.compressAndSaveBitmapToSDCard(newBitmap, "xx100.jpg", 80);
      //问题:
      //原图大小为625x690 90.2kB
      //如果设置图片500x500 压缩后大小为171kB.即压缩后kB反而变大了.
      //原因是将:compress(Bitmap.CompressFormat.JPEG, quality, fileOutputStream);
      //第二个参数quality设置得有些大了(比如100).
      //常用的是80,刚设100太大了造成的.
      return newBitmap;
  }

  /**
   * 变形
   * @param bitmap
   * @return
   */
  public Bitmap postSkewBitmap(Bitmap bitmap){
      // 得到图片原始的高宽
      int rawHeight = bitmap.getHeight();
      int rawWidth = bitmap.getWidth();
      Matrix matrix = new Matrix();
      // 设置图片的倾斜
      matrix.postSkew(0.1f, -0.1f);
      Bitmap newBitmap = Bitmap.createBitmap(bitmap, 0, 0, rawWidth,rawWidth, matrix, true);
      return newBitmap;
  }

  /**
   * 旋转
   * @param bitmap
   * @return
   */
  public Bitmap rotateBitmap(Bitmap bitmap){
      // 得到图片原始的高宽
      int rawHeight = bitmap.getHeight();
      int rawWidth = bitmap.getWidth();
      Matrix matrix = new Matrix();
      // 设置图片的旋转角度
      matrix.postRotate(-45);
      Bitmap newBitmap = Bitmap.createBitmap(bitmap, 0, 0, rawWidth,rawWidth, matrix, true);
      return newBitmap;
  }

合成两张bitmap

  • 为bitmap设置圆角

    public Bitmap toRoundCorner(Bitmap bitmap,int pixels){

        Bitmap roundCornerBitmap = Bitmap.createBitmap(bitmap.getWidth(), bitmap.getHeight(), Config.ARGB_8888);
    Canvas canvas = new Canvas(roundCornerBitmap);
    int color = 0xff424242;//int color = 0xff424242;
    Paint paint = new Paint();
    paint.setColor(color);
    //防止锯齿
    paint.setAntiAlias(true);
    Rect rect = new Rect(0, 0, bitmap.getWidth(), bitmap.getHeight());
    RectF rectF = new RectF(rect);
    float roundPx = pixels;
    //相当于清屏
    canvas.drawARGB(0, 0, 0, 0);
    //先画了一个带圆角的矩形
    canvas.drawRoundRect(rectF, roundPx, roundPx, paint);
    paint.setXfermode(new PorterDuffXfermode(Mode.SRC_IN)); ---------------------------此处注意Mode具体参考下方文章链接
    //再把原来的bitmap画到现在的bitmap！！！注意这个理解
    canvas.drawBitmap(bitmap, rect, rect, paint);
    return roundCornerBitmap;
}

    参考文章：android 画图之setXfermode

    public Bitmap addbackgroundonlyicon(Bitmap bitmap_back, Bitmap bitmap_front) {
     Bitmap b1 =  bitmap_back;
     Bitmap b2 =  bitmap_front;
     if (!b1.isMutable()) { //设置图片为背景为透明
         b1 = b1.copy(Bitmap.Config.ARGB_8888, true);
     }
     Paint paint = new Paint();
     Canvas canvas = new Canvas(b1);
     canvas.drawBitmap(b2, 0, 0, paint);
     canvas.save(Canvas.ALL_SAVE_FLAG);
     canvas.restore();
     return b1;
 }
